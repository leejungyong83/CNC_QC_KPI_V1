```python
# 번역 함수 사용법
from utils.language_manager import t

# 기본 사용
st.title(f"📊 {t('CNC QC KPI 대시보드')}")

# 메뉴에 적용
st.button(f"📝 {t('검사데이터입력')}", key="inspection_input")

# 메시지에 적용
st.success(t("로그인 성공!"))
```
